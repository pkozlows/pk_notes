\begin{frame}{Cumulant expansion}
For a quasiparticle indexed by $p$,
\begin{align} G^R_{pp}(t) = G_{0,pp}^R(t) e^{C_{pp}^R(t)}, \label{eq:cumulant_G}\end{align} where \begin{align} C_{pp}^R(t) = -\frac{1}{\pi}\int d\omega \frac{\mathrm{Im} \Sigma_{pp}^R(\omega)}{\omega^2} \left(e^{-i\omega t} + i \omega t -1\right). \label{eq:cumulant}\end{align}
\pause
I will explore three different approximations for $\Sigma(\omega)$ : GW, PT2, and BWs2.
\end{frame}

  \Sigma ^{GW}(\omega ) = G^{HF}(\omega)^{-1} - G(\omega)^{-1}
\end{equation}

\section{Future directions}

\begin{frame}{Self-consistent cumulant}
\begin{enumerate}
\item Solve for the cumulant via
\begin{align} C_{pp}^R(t) = -\frac{1}{\pi}\int d\omega \frac{\mathrm{Im} \Sigma_{pp}^R(\omega)}{\omega^2} \left(e^{-i\omega t} + i \omega t -1\right). \label{eq:cumulant}\end{align}
\pause
\item Solve for the Green's function via
\begin{align} G^R_{pp}(t) = G_{0,pp}^R(t) e^{C_{pp}^R(t)}. \label{eq:cumulant_G}\end{align}
\pause
\item Solve for the self-energy via
\begin{align}
\Sigma ^R(\omega ) &= G_0^R(\omega)^{-1} - G^R(\omega)^{-1} \label{eq:rearranged_dyson}
\end{align}
and plug into equation~\ref{eq:cumulant}.
\pause
\item Iterate until self-consistency
\end{enumerate}
\end{frame}
\begin{frame}{Memory kernel formulation}
The GQME describes the time evolution of a correlation function $\mathcal{C}(t)$ as
\begin{equation}
\dot{\mathcal{C}}(t) = \mathcal{C}(t) {\Omega_1} - \int_{0}^{t} d \tau\, \mathcal{C}(t - \tau) \mathcal{K}_1(\tau) + D(t),
\label{eq:gqme}
\end{equation}
\pause
The equation of motion for the Green's function is
\begin{equation}
i\partial_t G^R(t) =  \epsilon_0 G^R(t)  + \int_0^t d\tau\, \Sigma^R(t - \tau)\, G^R(\tau),
% (i\partial_t - \epsilon_0) G^R(t) - \int_0^t d\tau\, \Sigma^R(t - \tau)\, G^R(\tau) = 0 \label{eq:eom_retarded}
